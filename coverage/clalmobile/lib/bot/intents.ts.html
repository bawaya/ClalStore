export function detectLanguage(text: string): "ar" | "he" | "en" {
  if (HE_PATTERN.test(text)) return "he";

  // Check for English keywords
  if (/^[a-zA-Z0-9\s.,!?'"()-]+$/.test(text)) return "en";
  return "ar"; // default
}
